Venn diagram demo

      venn-beta
        title Very Basic Venn Diagram
        set A
        set B
        set A,B   label: AB
      
      venn-beta
        title Three overlapping sets
        set A
        set B
        set C
        set A,B     label: AB,   background: skyblue
        set B,C     label: BC,   background: orange
        set A,C     label: AC,   background: lightgreen
        set A,B,C   label: ABC,  background: white,      color: red
      
      %%{init: { 'venn': { 'debugTextLayout': true } } }%%
      venn-beta
        title Four sets with all pairwise intersections
        set A      size: 18
        set B      size: 18
        set C      size: 18
        set D      size: 18
        set A,B    size: 5
        set A,C    size: 2.5
        set A,D    size: 5
        set B,C    size: 5
        set B,D    size: 2.5
        set C,D    size: 5
      
      venn-beta
        title Three concatenated sets and one isolated set
        set A    size: 15,  label: ALPHA
        set B    size: 5,   label: BETA
        set C    size: 10
        set D    size: 10,  label: DELTA
        set A,B  size: 1,   label: AB,    background: gold
        set B,D  size: 1
      
      venn-beta
