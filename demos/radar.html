        title My favorite ninjas
        axis Agility, Speed, Strength
        axis Stam["Stamina"] , Intel["Intelligence"]
      
        curve Ninja1["Naruto"]{
            Agility 2, Speed 2,
            Strength 3, Stam 5,
            Intel 0
        }
        curve Ninja2["Sasuke"]{2, 3, 4, 1, 5}
        curve Ninja3["Ninja"] {3, 2, 1, 5, 4}
      
        showLegend true
        ticks 3
        max 8
        min 0
        graticule circle
